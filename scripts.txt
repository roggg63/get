#!/usr/bin/env python3
# -*- coding: utf-8 -*-

import RPi.GPIO as GPIO
import time

class R2R_ADC:
    """
    8-битный АЦП последовательного счёта на основе R-2R ЦАП и компаратора.
    """

    def __init__(self, dynamic_range, compare_time=0.01, verbose=False):
        """
        Конструктор класса.

        :param dynamic_range: максимальное напряжение на выходе ЦАП (В)
        :param compare_time: время ожидания после установки числа на ЦАП (с)
        :param verbose: флаг вывода отладочной информации
        """
        self.dynamic_range = dynamic_range
        self.verbose = verbose
        self.compare_time = compare_time

        # Пины для управления R-2R ЦАП (8 линий, порядок — от младшего бита к старшему)
        self.bits_gpio = [26, 20, 19, 16, 13, 12, 25, 11]
        # Пин, подключённый к выходу компаратора
        self.comp_gpio = 21

        GPIO.setmode(GPIO.BCM)
        GPIO.setup(self.bits_gpio, GPIO.OUT, initial=GPIO.LOW)
        GPIO.setup(self.comp_gpio, GPIO.IN)

    def number_to_dac(self, number):
        """
        Подаёт число на выход ЦАП, устанавливая состояния выводов GPIO.

        :param number: целое число от 0 до 255
        """
        if self.verbose:
            print(f"Подача числа {number} на ЦАП")
        for i, pin in enumerate(self.bits_gpio):
            # i-й бит числа (0 = младший бит)
            bit = (number >> i) & 1
            GPIO.output(pin, bit)

    def sequential_counting_adc(self):
        """
        Выполняет аналого-цифровое преобразование методом последовательного счёта.

        :return: целое число (0..255), соответствующее входному напряжению
        """
        for code in range(256):
            self.number_to_dac(code)
            time.sleep(self.compare_time)

            # Компаратор выдаёт 1, когда V_dac > V_in
            if GPIO.input(self.comp_gpio) == 1:
                if self.verbose:
                    print(f"Превышение достигнуто при коде {code}")
                return code

        # Если ни при одном коде не превысило, возвращаем максимум
        if self.verbose:
            print("Превышения не достигнуто, вернём 255")
        return 255

    def get_sc_voltage(self):
        """
        Измеряет напряжение на входе АЦП.

        :return: напряжение в вольтах
        """
        code = self.sequential_counting_adc()
        voltage = code * self.dynamic_range / 255.0
        if self.verbose:
            print(f"Измеренный код: {code}, напряжение: {voltage:.3f} В")
        return voltage

    def cleanup(self):
        """
        Очистка GPIO: обнуление выходов ЦАП и сброс настроек.
        """
        self.number_to_dac(0)
        GPIO.cleanup(self.bits_gpio + [self.comp_gpio])
        if self.verbose:
            print("GPIO очищены")

    def __del__(self):
        """
        Деструктор, вызывается при удалении объекта.
        """
        self.cleanup()


if __name__ == "__main__":
    # Измеренное мультиметром значение динамического диапазона (В)
    DYNAMIC_RANGE = 3.30   # замените на реальное значение вашей платы

    try:
        # Создаём объект АЦП (verbose=True для отладки)
        adc = R2R_ADC(dynamic_range=DYNAMIC_RANGE, compare_time=0.01, verbose=False)

        print("Начинаем измерение напряжения. Для выхода нажмите Ctrl+C\n")

        while True:
            voltage = adc.get_sc_voltage()
            print(f"Напряжение: {voltage:.3f} В")
            time.sleep(0.2)   # небольшая пауза между измерениями

    except KeyboardInterrupt:
        print("\nПрерывание по запросу пользователя")

    finally:
        # Явный вызов очистки GPIO
        if 'adc' in locals():
            adc.cleanup()
